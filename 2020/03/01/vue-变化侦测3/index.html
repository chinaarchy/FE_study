# $watch

$watch的用法如下：

vm.$watch(expOrFn, callback, [options])
1

$watch是对watcher的一种封装，为了满足$watch的需求，需要对watcher进行实现

Vue.prototype.$watch = function (expOrFn, cb, options) {
    const vm = this
    options = options || {}
    const watcher = new Watcher(vm, expOrFn, cb, options)
    if (options.immediate){
        cb.call(vm, watcher.value)
    } 
    return function unwatch() {
        watcher.teardown()
    }
}

逻辑比较清楚，但是参数中的expOrFn可能为函数，故需要对Watcher进行升级，可以处理函数类型。

同时还应该创建teardown函数来移除订阅，故还需要在Watcher中完善对应的收集等函数。

class Watcher {
    constructor(vm, expOrFn, cb) {
        this.vm = vm
        this.deps = []
        this.depIds = new Set()
        if(typeof expOrFn === 'function'){
            this.getter = expOrFn
        }else{
            this.getter = parsePath(expOrFn)
        }
        
    }
    
    
    addDep(dep){
        const id = dep.id
        if(!this.depIds.has(id)){
            this.depIds.add(id)
            this.deps.push(dep)
            dep.addSub(this)
        }
    }
    teardown(){
        let i = this.deps.length
        while (i--){
            this.deps[i].removeSub(this)
        } 
    }

}

在Watcher中添加DepIds和Deps收集依赖，其中depIds使用了Set构建保证不会重复的监控值，出现循环调用的问题。

下面完成对参数deep的实现，在Watcher中getter中添加对应的依赖添加逻辑。

const seenObjects = new Set()

function traverse(val) {
    _traverse(val, seenObjects)
    seenObjects.clear()
}

function _traverse(val, seen) {
    let i, keys
    const isA = Array.isArray(val)
    if ((!isA && !isObject(val)) || Object.isFrozen(val)) return
    if (val.__ob__) {
        const depId = val.__ob__.dep.id
        if (seen.has(depId)) {
            return
        }
        seen.add(depId)
    }
    if (isA) {
        i = val.length
        while (i--) _traverse(val[i], seen)
    } else {
        keys = Object.keys(val)
        i = keys.length
        while (i--) _traverse(val[keys[i]], seen)
    }
}

seenObject储存对应的依赖的id值，首先判断不是数组、根数据或者被冻结的数据，返回。 之后判断是否有__ob__，将对应的依赖判断放入seenObject中，最后之前没有被监测， 分为数组和对象分别递归调用。

# $set

另一个较常用的api是$set设置对象中的数据，使其可以被监测到。

思路上首先处理数组的情况，之后处理参数已经存在在target中的情况，最后处理新增的属性。

Vue.prototype.$set(target, key, val) 
{
    if (Array.isArray(target) && isValidArrayIndex(key)) {
        target.length = Math.max(target.length, key)
        target.splice(key, 1, val)
        return val
    }
    
    if(key in target && !(key in Object.prototype)){
        target[key] = val
        return val
    }
    
    const ob = target.__ob__
    if (target._isVue || (ob && ob.vmCount)){
        warn()
        return val
    }
    if(!ob){
        target[key] = val
        return val
    }
    defineReactive(ob.value, key, val)
    ob.dep.notify()
    return val
}

# $delete

$delete的实现思路可以用一下简单的思路进行实现

delete this.obj.name
this.obj.__ob__.dep.notify()
1
2

在实现上基本和$set相同，首先判断如果为数组，则直接通过splice进行修改，会直接通知进行了修改， 之后如果数据为Vue数据或者不存在数据直接推出，最后如果数据不是响应式数据，就直接进行修改。

Vue.prototype.$delete = function (target, key) {
    if (Array.isArray(target) && isValidArrayIndex(key)){
        target.splice(key, 1)
        return 
    } 
    const ob = target.__ob__
    if(target._isVue || (ob && ob.vmCount)){
        warn()
        return
    }
    if(!hasOwn(target, key)){
        return
    }    
    delete target[key]
    if (!ob){
        return
    } 
    ob.deps.notify()
}
